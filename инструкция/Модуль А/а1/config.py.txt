from pydantic_settings import BaseSettings, SettingsConfigDict
from pydantic import Field

class Settings(BaseSettings):
    model_config = SettingsConfigDict(env_file=".env", env_file_encoding="utf-8")

    sqlite_path: str = Field(default="data/app.sqlite3")
    data_ttl_hours: int = Field(default=24)

    # Тайлы: топо и спутник
    topo_tile_url: str = Field(default="https://tile.openstreetmap.org/{z}/{x}/{y}.png")
    sat_tile_url: str = Field(
        default="https://server.arcgisonline.com/ArcGIS/rest/services/"
                "World_Imagery/MapServer/tile/{z}/{y}/{x}"
    )
    tile_zoom: int = Field(default=13)
    radius_m: int = Field(default=500)

settings = Settings()
